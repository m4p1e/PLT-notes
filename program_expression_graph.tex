\subsection{Definition}

\begin{example}
\rm 给定代码片段
$$
\begin{aligned}
&e_1 = \&x \\
&e_2 = \&x \\
&e_3 = e_2 \\
\end{aligned}
$$
推出$V(e_3,e_1)$.

我们直接得到的$A$边和$D$边有
$$
\begin{aligned}
&A(\&x,e_1) \\
&A(\&x,e_2) \\
&A(e_2,e_2) \\
&D(\&x, x) \\
&D(e_1, *e_2) \\
&D(e_2, *e_2) \\
&D(e_3, *e_3) \\
\end{aligned}
$$

\end{example}
